# 全科学习进度总览

> 每完成一个知识点，回到对应文件把 `status` 改为 `已完成` ✅

```dataviewjs
const subjects = [
    { name: '数学·高数', path: '"02-科目/数学/知识点/高数"' },
    { name: '数学·线代', path: '"02-科目/数学/知识点/线代"' },
    { name: '政治·马原', path: '"02-科目/政治/知识点/马原"' },
    { name: '政治·毛概', path: '"02-科目/政治/知识点/毛概"' },
    { name: '政治·史纲', path: '"02-科目/政治/知识点/史纲"' },
    { name: '政治·思修', path: '"02-科目/政治/知识点/思修"' },
    { name: '843·概率论', path: '"02-科目/专业课/知识点/技术/概率论"' },
    { name: '843·数理统计', path: '"02-科目/专业课/知识点/技术/数理统计"' },
    { name: '843·数据结构', path: '"02-科目/专业课/知识点/技术/数据结构"' },
    { name: '843·计算机网络', path: '"02-科目/专业课/知识点/技术/计算机网络"' },
    { name: '843·操作系统', path: '"02-科目/专业课/知识点/技术/操作系统"' },
    { name: '843·硬件基础', path: '"02-科目/专业课/知识点/技术/硬件基础"' },
    { name: '843·设计方向', path: '"02-科目/专业课/知识点/设计"' },
];

const statusIcon = s => s==='已完成'?'✅':s==='进行中'?'🔄':s==='待复习'?'🔁':'⬜';
const bar = (done,total) => {
    if(!total) return '—';
    const p=Math.round(done/total*8);
    return '█'.repeat(p)+'░'.repeat(8-p)+` ${done}/${total}`;
};

let rows = [];
for(const sub of subjects){
    const pages = dv.pages(sub.path);
    const done = pages.filter(p=>p.status==='已完成').length;
    const inprog = pages.filter(p=>p.status==='进行中').length;
    const total = pages.length;
    rows.push([sub.name, bar(done, total), done, inprog, total - done - inprog]);
}
dv.table(['模块', '进度', '✅完成', '🔄进行中', '⬜未开始'], rows);
```
