## 错题章节热图（自动聚合）

```dataviewjs
// 聚合全科目错题，按章节统计频次
const allErrors = dv.pages('"02-科目"')
  .where(p => p.错误类型 && p.章节);

const byChapter = {};
const bySubject = {};

for (const p of allErrors) {
  const ch = p.章节 || "未分类";
  const sub = p.subject || "未知";
  byChapter[ch] = (byChapter[ch] || 0) + 1;
  bySubject[sub] = (bySubject[sub] || 0) + 1;
}

// 按科目分布
dv.header(3, "按科目错题分布");
dv.table(["科目", "错题数", "占比"],
  Object.entries(bySubject)
    .sort((a,b) => b[1]-a[1])
    .map(([sub, cnt]) => [
      sub, cnt,
      "█".repeat(Math.round(cnt/allErrors.length*10)) + ` ${(cnt/allErrors.length*100).toFixed(0)}%`
    ])
);

// 高频章节 Top 10
dv.header(3, "高频错误章节 Top 10");
dv.table(["章节", "错题数"],
  Object.entries(byChapter)
    .sort((a,b) => b[1]-a[1])
    .slice(0, 10)
    .map(([ch, cnt]) => [ch, "█".repeat(Math.min(cnt,10)) + ` ${cnt}`])
);

if (allErrors.length === 0) {
  dv.paragraph("📝 暂无错题记录。使用模板 `错题记录` 新建第一道错题后，此处自动显示。");
}
```
